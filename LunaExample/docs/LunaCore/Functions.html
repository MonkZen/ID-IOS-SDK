    NSError* bestShotNSErrorFrom(BestShotError error, NSError *nsError) {
    if (nsError) {
        return nsError;
    }
    switch (error) {
        case BAD_QUALITY: {
            return [NSError errorWithDomain: kBestShotErrorDomain
                                       code: BAD_QUALITY
                                   userInfo: @{ NSLocalizedDescriptionKey: [@"errors.bad_quality" localized] }];
        }
        case BAD_HEAD_POSE: {
            return [NSError errorWithDomain: kBestShotErrorDomain
                                       code: BAD_HEAD_POSE
                                   userInfo: @{ NSLocalizedDescriptionKey: [@"errors.bad_head_pose" localized] }];
        }
        case MULTIPLE_FACES: {
            return [NSError errorWithDomain: kBestShotErrorDomain
                                       code: MULTIPLE_FACES
                                   userInfo: @{ NSLocalizedDescriptionKey: [@"errors.multiple_faces_detected" localized] }];
        }
        case LIVENESS_ERROR: {
            return [NSError errorWithDomain: kBestShotErrorDomain
                                       code: LIVENESS_ERROR
                                   userInfo: @{ NSLocalizedDescriptionKey: [@"errors.liveness_error" localized] }];
        }
        case TRACK_END: {
            return [NSError errorWithDomain: kBestShotErrorDomain
                                       code: TRACK_END
                                   userInfo: @{ NSLocalizedDescriptionKey: [@"errors.track_end" localized] }];
        }
    }
}
